  bucket: thanos
  endpoint: 127.0.0.1:9000
  access_key: minioadmin
  secret_key: minioadmin
  insecure: true
  signature_version2: false
  http_config:
    idle_conn_timeout: 5m
    response_header_timeout: 1m
    insecure_skip_verify: true
